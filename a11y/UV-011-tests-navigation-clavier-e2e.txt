Fixtures à créer (e2e/fixtures/keyboard.ts):
```typescript
export const keyboardTest = async ({ page }) => {
  return {
    tab: () => page.keyboard.press('Tab'),
    shiftTab: () => page.keyboard.press('Shift+Tab'),
    enter: () => page.keyboard.press('Enter'),
    space: () => page.keyboard.press('Space'),
    esc: () => page.keyboard.press('Escape'),
    arrows: (direction: 'up' | 'down' | 'left' | 'right') => 
      page.keyboard.press(`Arrow${direction}`),
    getFocusedElement: () => page.evaluate(() => 
      document.activeElement?.tagName
    ),
  }
}
```

Tests à implémenter:
```typescript
test('Modal - Keyboard navigation', async ({ page }) => {
  const kb = keyboardTest(page)
  
  // Open modal
  await page.click('[data-testid="open-modal"]')
  
  // Focus trapped in modal
  await kb.tab()
  await expect(page.locator('[role="dialog"]')).toBeFocused()
  
  // Close with Esc
  await kb.esc()
  await expect(page.locator('[role="dialog"]')).not.toBeVisible()
})
```
