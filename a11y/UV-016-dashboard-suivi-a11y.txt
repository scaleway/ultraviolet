Script de génération (utils/scripts/a11y-dashboard.ts):
```typescript
const generateDashboard = async () => {
  const currentMetrics = await scanAllComponents()
  const historicalMetrics = loadHistoricalData()
  
  const dashboard = {
    summary: calculateSummary(currentMetrics),
    trend: calculateTrend(currentMetrics, historicalMetrics),
    topViolations: getTopViolations(currentMetrics),
    improvements: getImprovements(currentMetrics, historicalMetrics),
  }
  
  writeToFile('docs/a11y-dashboard.md', formatDashboard(dashboard))
  updateBadge('a11y-badge.svg', dashboard.wcagCompliance)
}
```
