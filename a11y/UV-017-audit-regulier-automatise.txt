GitHub Action (.github/workflows/a11y-weekly.yml):
```yaml
name: Weekly A11y Audit
on:
  schedule:
    - cron: '0 9 * * 1'  # Every Monday at 9am UTC
  workflow_dispatch:  # Allow manual trigger

jobs:
  a11y-audit:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Setup
        run: |
          pnpm install
          pnpm build
          pnpm build:storybook
      - name: Start Storybook
        run: pnpm http-server ./storybook-static -p 6006 &
      - name: Run A11y Audit
        run: pnpm tsx utils/scripts/a11y-full-scan.ts
      - name: Generate Report
        run: pnpm tsx utils/scripts/generate-a11y-report.ts
      - name: Upload Report
        uses: actions/upload-artifact@v4
        with:
          name: a11y-weekly-report
          path: a11y-report.html
      - name: Check for Regressions
        run: pnpm tsx utils/scripts/check-a11y-regressions.ts
      - name: Create Issue if Regression
        if: failure()
        uses: actions/github-script@v6
        with:
          script: |
            github.rest.issues.create({
              owner: context.repo.owner,
              repo: context.repo.repo,
              title: '🚨 A11y Regression Detected',
              body: 'Weekly audit found new violations...',
              labels: ['a11y', 'regression', 'critical']
            })
```
